## 3. 核心模块设计

### 3.1 事件总线（EventBus）

```go
// EventBus 事件总线
type EventBus struct {
    subscribers map[string][]chan Event
    mu          sync.RWMutex
}

type Event struct {
    Type    string                 // "alert"|"dispatch"|"plan"|"command"
    Source  string                 // "kj90x"|"manual"|"ai"
    Data    map[string]interface{}
    TraceID string                 // 全链路追踪ID
    Time    time.Time
}

// 订阅示例
eb.Subscribe("alert", ch)
eb.Publish(Event{Type: "alert", Source: "kj90x", Data: {...}})
```

## 4. 数据库设计

```sql
-- 报警记录
CREATE TABLE alerts (
    id TEXT PRIMARY KEY,
    source TEXT NOT NULL,       -- kj90x/manual/ai
    level INTEGER,              -- 1-5风险等级
    location TEXT,
    gas_type TEXT,
    value REAL,
    status TEXT DEFAULT 'open', -- open/acknowledged/resolved
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    acknowledged_at TIMESTAMP,
    resolved_at TIMESTAMP
);

-- 预案模板
CREATE TABLE plans (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    trigger_type TEXT,          -- gas_overflow/fire/collapse/manual
    steps JSONB NOT NULL,        -- 预案步骤
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 指令记录
CREATE TABLE commands (
    id TEXT PRIMARY KEY,
    plan_id TEXT REFERENCES plans(id),
    alert_id TEXT REFERENCES alerts(id),
    type TEXT NOT NULL,
    target TEXT NOT NULL,
    content TEXT,
    status TEXT DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    executed_at TIMESTAMP
);

-- 知识库
CREATE TABLE knowledge (
    id TEXT PRIMARY KEY,
    category TEXT,              -- accident/regulation/equipment
    title TEXT NOT NULL,
    content TEXT,
    tags TEXT[],
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 调度日志
CREATE TABLE dispatch_logs (
    id TEXT PRIMARY KEY,
    operator TEXT,
    action TEXT,
    target TEXT,
    result TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```
